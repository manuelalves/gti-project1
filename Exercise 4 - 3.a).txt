

declare default element namespace "http://www.parlamento.pt";
let $doc := doc("/Users/jessicaribeiro/Desktop/IST/1semestre/GTI/Projecto/gti-project1/parliament-data-small.xml")

for $politician in ($doc//politician)
let $count := count($doc//speech[@politician eq $politician/@code])
order by $count descending
return 
  copy $p := $politician
  modify (
   insert node ( attribute num-interventions \{$count\})
   into $p )
  return  <politician> <name> \{$p/text()\} </name> <interventions> \{$p/@num-interventions\} </interventions> </politician>}
